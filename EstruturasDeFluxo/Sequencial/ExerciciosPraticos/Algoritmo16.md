Algoritmo "Q16 - exSequencial"

Var

x, y, aux: real

Inicio

escreva("Insira o valor de x: ")

leia(x)

escreva("Insira o valor de y: ")

leia(y)

escreval

escreval("Valores antes da troca: ")

escreval("x =", x,", y =", y)

aux <- x

x <- y

y <- aux

escreval

escreval("Valores depois da troca: ")

escreval("x =", x,", y =", y)

Fimalgoritmo
